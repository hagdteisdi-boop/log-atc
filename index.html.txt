    <script>
        // Key mặc định ban đầu nếu chưa có admin
        if(!localStorage.getItem('titan_database')) {
            const initialKeys = [
                {key: 'sytuan123', expiry: Date.now() + 86400000 * 30}, // 30 ngày
                {key: '1dayvip1', expiry: Date.now() + 86400000},       // 1 ngày
                {key: 'Be1aa1', expiry: Date.now() + 3600000 * 2}      // 2 giờ
            ];
            localStorage.setItem('titan_database', JSON.stringify(initialKeys));
        }

        function login() {
            const input = document.getElementById('key-input').value.trim();
            const db = JSON.parse(localStorage.getItem('titan_database') || '[]');
            const now = Date.now();
            
            // Lọc bỏ key hết hạn ngay khi log
            const freshDb = db.filter(item => item.expiry > now);
            localStorage.setItem('titan_database', JSON.stringify(freshDb));

            const foundKey = freshDb.find(item => item.key === input);

            if(foundKey) {
                document.getElementById('login-box').style.display = 'none';
                document.getElementById('main-app').style.display = 'block';
                startApp(foundKey.expiry);
            } else {
                document.getElementById('msg').innerText = "❌ Key không đúng hoặc đã hết hạn!";
            }
        }

        function startApp(expiryTime) {
            // Tính thời gian còn lại
            setInterval(() => {
                const diff = expiryTime - Date.now();
                if(diff <= 0) location.reload();
                const days = Math.floor(diff / 86400000);
                const hours = Math.floor((diff % 86400000) / 3600000);
                const mins = Math.floor((diff % 3600000) / 60000);
                document.getElementById('time-left').innerText = `Hạn: ${days}n ${hours}g ${mins}p`;
            }, 1000);

            // Quét IP
            fetch('https://api.ipify.org?format=json').then(r => r.json()).then(d => document.getElementById('ip').innerText = d.ip);
            
            // Thông tin máy
            const modelName = navigator.userAgent.split('(')[1].split(')')[0].split(';')[1] || "Thiết bị Android";
            document.getElementById('model').innerText = modelName.trim();
            document.getElementById('ram').innerText = (navigator.deviceMemory || "8") + " GB (Ước tính)";
            document.getElementById('resolution').innerText = screen.width + "x" + screen.height;

            // Pin
            if(navigator.getBattery) {
                navigator.getBattery().then(b => {
                    document.getElementById('battery').innerText = Math.round(b.level * 100) + "% " + (b.charging ? "⚡" : "🔋");
                });
            }

            // Giả lập quét Root (Vì trình duyệt không thể truy cập file hệ thống thật)
            setTimeout(() => {
                document.getElementById('root-result').innerText = `✖ File ROOT của thiết bị ${modelName.trim()} KHÔNG TỒN TẠI`;
            }, 2000);
        }
    </script>
